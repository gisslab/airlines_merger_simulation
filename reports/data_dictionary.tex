\section{Data Construction Process}

This dataset was constructed by processing quarterly DB1B Market \textbf{10\% sample} files from the U.S.\ Department of Transportation's Bureau of Transportation Statistics and merging with auxiliary data on airport characteristics, demographics, and market structure. The DB1B 10\% sample is the standard public-use file provided by BTS.

\subsection{Data Sources}
\begin{itemize}
\item \textbf{Primary}: U.S. DOT Bureau of Transportation Statistics, Airline Origin and Destination Survey (DB1B Market)
\item \textbf{Website}: \url{https://www.transtats.bts.gov/DatabaseInfo.asp?DB_ID=125}
\item \textbf{Auxiliary}: Census population data, airport characteristics, vacation destinations
\end{itemize}

\subsection{Processing Steps}
\begin{enumerate}
\item \textbf{Market Definition}: Origin-destination city pairs based on city market IDs
\item \textbf{Time Aggregation}: Quarterly data aggregated from individual trip records
\item \textbf{Geographic Scope}: Limited to Continental U.S. (CONUS) airports
\item \textbf{Market Size Filter}: Markets with $\geq$20 passengers/day on average ($\geq$182.5 quarterly passengers in 10\% sample)
\item \textbf{Price Filter}: Real fares between \$25-\$2,500 (2019 dollars)
\item \textbf{Competition Filter}: Excludes monopoly markets (rival\_carriers $>$ 0)
\end{enumerate}

\subsection{Dataset Overview}
\begin{itemize}
\item \textbf{File}: \texttt{airline\_data\_main.dta}
\item \textbf{Period}: 2005 Q1 - 2019 Q4 (60 quarters)
\item \textbf{Level}: Product-market-time (carrier-route-quarter)
\item \textbf{Observations}: 1,677,867 product-market-quarter combinations
\item \textbf{Markets}: 8,059 unique origin-destination city pairs
\item \textbf{Carriers}: 134 unique airline codes
\end{itemize}

\textbf{Variable} & \textbf{Type} & \textbf{Description} \\
\midrule
\endhead

\texttt{total\_passengers} & integer & Total quarterly passengers for this carrier on this route \\
\texttt{average\_fare} & float & Average ticket price in 2019 real dollars. Constructed as passenger-weighted average of individual fares \\

\subsection{Competition and Network Structure}

\begin{longtable}[c]{p{5.1cm}p{1.8cm}p{7.7cm}}
\toprule
\textbf{Variable} & \textbf{Type} & \textbf{Description} \\
\midrule
\endhead

% \multicolumn{3}{l}{\textbf{Market Competition Measures}} \\
\texttt{fringe\_carriers} & integer & Number of fringe carriers serving this market in this quarter \\
\texttt{rival\_carriers} & integer & Number of rival carriers in market (total carriers minus focal carrier) \\
\texttt{presence} & float & Carrier's passenger share at origin airport. Calculated as carrier\_passengers\_at\_origin / total\_passengers\_at\_origin \\
\texttt{total\_presence} & float & Sum of all carriers' presence measures in this market \\

% \multicolumn{3}{l}{\textbf{Network Size Variables}} \\
\texttt{num\_destinations} & integer & Number of destinations served by carrier from this origin airport in this quarter \\
\texttt{num\_markets} & integer & Number of distinct origin-destination markets served by carrier in this quarter \\
\texttt{total\_num\_destinations} & integer & Total destinations served by all carriers from this origin \\
\texttt{total\_num\_markets} & integer & Total markets served by all carriers \\

% \multicolumn{3}{l}{\textbf{Rival Carrier Characteristics}} \\
\texttt{average\_distance\_rival} & float & Average distance flown by rival carriers in this market. Calculated as leave-one-out mean: $\frac{\sum \text{rival\_distance} - \text{own\_distance}}{N-1}$ \\
\texttt{average\_presence\_rival} & float & Average presence of rival carriers at origin airport. Leave-one-out mean of presence measures \\
\texttt{average\_num\_destinations\_rival} & float & Average network size of rival carriers (destinations). Leave-one-out mean \\
\texttt{average\_num\_markets\_rival} & float & Average network size of rival carriers (markets). Leave-one-out mean \\

\bottomrule
\end{longtable}

\subsection{Missing Data}
\begin{itemize}
\item Population data: $\sim$1.25\% missing (primarily small airports/MSAs)
\item Core variables: 99\%+ coverage for passengers, fares, distance
\item Complete temporal coverage: All quarters 2005--2019
\end{itemize}
